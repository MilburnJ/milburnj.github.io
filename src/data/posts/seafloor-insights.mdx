# Sample U‑Net model in Keras
from tensorflow.keras import layers, Model

def build_unet(input_shape=(224,224,3), num_classes=17):
    inputs = layers.Input(shape=input_shape)

    # Encoder
    c1 = layers.Conv2D(64, 3, activation='relu', padding='same')(inputs)
    p1 = layers.MaxPooling2D()(c1)

    # Bottleneck
    bn = layers.Conv2D(128, 3, activation='relu', padding='same')(p1)

    # Decoder
    u1 = layers.UpSampling2D()(bn)
    outputs = layers.Conv2D(num_classes, 1, activation='softmax')(u1)

    return Model(inputs, outputs)

model = build_unet()
model.compile(
    optimizer='adam',
    loss='categorical_crossentropy',
    metrics=['accuracy']
)
